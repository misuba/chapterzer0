def rename(old,new)
  nil if (old.empty? || new.empty?) 
  @resource = "tags/rename?old=#{old}&new=#{new}"
  @request  = request(URI.escape(@resource))
  @doc      = (Document.new @request).root

  true if @doc.text == "done"
end
